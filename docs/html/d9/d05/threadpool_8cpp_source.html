137void TP::init() {

140 }

150void TP::stop() {
151 if (m_started == 0) {
152 return;
153 }
154 m_fstop = 0;
155 m_faccept = 0;
156 wait_empty();
157
158 pthread_mutex_lock(&m_q_mtx);
159 m_fstop = 1;
160 pthread_cond_broadcast(&m_q_cond);
161 pthread_mutex_unlock(&m_q_mtx);
162 // Detener los hilos
163 for (size_t i = 0; i < m_started; ++i) {
164 pthread_join(m_ths[i], nullptr);
165 }
166 m_started = 0;
167 m_nth = 0;

169void TP::add(uint32_t qid, void (*nfn)(void*), void* arg) {
170 if (m_faccept == 0) {

172 }
