135 orig -= sizeof (uint64_t);

427int
428read_chunk (void **dest, size_t *len, struct request *const req)
429{
430 if (req->iovec_count == 0)
431 {
432 return ERROR_CODE;
433 }
434
435 size_t max_iov_cont = 0; //< Total size of request
436 for (size_t i = 0; i < req->iovec_count; ++i)
437 {
438 max_iov_cont += req->iov[i].iov_len;
439 }
440 size_t cont_sz = 0;
441 size_t cont_rem = 0;
442 size_t curr_iov = 0;
443 size_t curr_iov_off = 0;
444 size_t dest_off = 0;
445 void *dest_ptr = NULL;
446 if (req->prev && req->prev_size && req->prev_remain)
447 {
448 cont_sz = req->prev_size;
449 cont_rem = req->prev_remain;
450 curr_iov = 0;
451 curr_iov_off = 0;
452 dest_off = cont_sz - cont_rem;
453 if (cont_rem <= max_iov_cont)
454 {
455 *dest = req->prev;
456 dest_ptr = *dest;
457 req->prev = NULL;
458 req->prev_size = 0;
459 req->prev_remain = 0;
460 }
461 else
462 {
463 dest_ptr = req->prev;
464 *dest = NULL;
465 }
466 }
467 else if (req->next_iov || req->next_offset)
468 {
469 curr_iov = req->next_iov;
470 curr_iov_off = req->next_offset;
471 cont_sz = *(
472 (size_t *)(((uint8_t *)req->iov[curr_iov].iov_base) + curr_iov_off));
473 cont_sz = ntohll (cont_sz);
474 curr_iov_off += sizeof (uint64_t);
475 cont_rem = cont_sz;
476 dest_off = cont_sz - cont_rem;
477 if ((curr_iov_off + cont_rem + 1) <= max_iov_cont)
478 {
479 *dest = malloc (cont_sz);
480 dest_ptr = *dest;
481 }
482 else
483 {
484 req->prev = malloc (cont_sz);
485 dest_ptr = req->prev;
486 *dest = NULL;
487 *len = 0;
488 }
489 }
490 else
491 {
492 curr_iov = 0;
493 curr_iov_off = 0;
494 cont_sz = *(
495 (size_t *)(((uint8_t *)req->iov[curr_iov].iov_base) + curr_iov_off));
496 cont_sz = ntohll (cont_sz);
497 curr_iov_off += sizeof (uint64_t);
498 cont_rem = cont_sz;
499 dest_off = cont_sz - cont_rem;
500 if (cont_rem <= max_iov_cont)
501 {
502 *dest = malloc (cont_sz);
503 dest_ptr = *dest;
504 }
505 else
506 {
507 req->prev = malloc (cont_sz);
508 dest_ptr = req->prev;
509 *dest = NULL;
510 }
511 }
512 size_t curr_iov_msg_rem = 0;
513
514 uint8_t ok = 1UL;
515 while (1)
516 {
517 curr_iov_msg_rem
518 = MIN (cont_rem, (req->iov[curr_iov].iov_len - curr_iov_off));
519 memcpy ((uint8_t *)dest_ptr + dest_off,
520 ((uint8_t *)req->iov[curr_iov].iov_base) + curr_iov_off,
521 curr_iov_msg_rem);
522 dest_off += curr_iov_msg_rem;
523 curr_iov_off += curr_iov_msg_rem;
524 cont_rem -= curr_iov_msg_rem;
525 if (cont_rem <= 0)
526 {
527 if (*(((uint8_t *)req->iov[curr_iov].iov_base) + curr_iov_off) != 0)
528 {
529 ok = 0UL;
530 }
531 *len = cont_sz;
532 ++curr_iov_off;
533 break;
534 }
535 if (curr_iov_off >= (req->iov[curr_iov].iov_len))
536 {
537 ++curr_iov;
538 if (curr_iov == req->iovec_count)
539 {
540 break;
541 }
542 curr_iov_off = 0;
543 }
544 }
545
546 if (req->prev)
547 {
548 req->prev_size = cont_sz;
549 req->prev_remain = cont_rem;
550 *dest = NULL;
551 len = 0;
552 }
553 else
554 {
555 req->prev_size = 0;
556 req->prev_remain = 0;
557 }
558 if (curr_iov < req->iovec_count)
559 {
560 uint64_t next_sz = *(uint64_t *)(((uint8_t *)req->iov[curr_iov].iov_base)
561 + curr_iov_off);
562 if ((req->iov[curr_iov].iov_len > curr_iov_off) && next_sz)
563 {
564 req->next_iov = curr_iov;
565 req->next_offset = curr_iov_off;
566 }
567 else
568 {
569 req->next_iov = 0;
570 req->next_offset = 0;
571 }
572 }
573
574 if (ok)
575 {
576 return SUCCESS_CODE;
577 }
578 free (dest_ptr);
579 dest_ptr = NULL;
580 *dest = NULL;
581 *len = 0;
582 req->next_iov = 0;
583 req->next_offset = 0;
584 for (size_t i = curr_iov; i < req->iovec_count; ++i)
585 {
586 for (size_t j = curr_iov_off; j < req->iov[i].iov_len; ++j)
587 {
588 uint8_t foot = *((uint8_t *)req->iov[i].iov_base) + j;
589 if (foot == 0)
590 {
591 req->next_iov = i;
592 req->next_offset = (j + 1) % req->iov[i].iov_len;
593 return ERROR_CODE;
594 }
595 }
596 }
597 return ERROR_CODE;
598}

669int
670saurion_set_socket (const int p)
671{
672 int sock = 0;
673 struct sockaddr_in srv_addr;
674
675 sock = socket (PF_INET, SOCK_STREAM, 0);
676 if (sock < 1)
677 {
678 return ERROR_CODE;
679 }
680
681 int enable = 1;
682 if (setsockopt (sock, SOL_SOCKET, SO_REUSEADDR, &enable, sizeof (int)) < 0)
683 {
684 return ERROR_CODE;
685 }
686
687 memset (&srv_addr, 0, sizeof (srv_addr));
688 srv_addr.sin_family = AF_INET;
689 srv_addr.sin_port = htons (p);
690 srv_addr.sin_addr.s_addr = htonl (INADDR_ANY);
691
692 if (bind (sock, (const struct sockaddr *)&srv_addr, sizeof (srv_addr)) < 0)
693 {
694 return ERROR_CODE;
695 }
696
697 if (listen (sock, ACCEPT_QUEUE) < 0)
698 {
699 return ERROR_CODE;
700 }
701
702 return sock;
703}

822static int
823saurion_worker_master_loop_it (struct saurion *const s,
824 struct sockaddr_in *client_addr,
825 socklen_t *client_addr_len)
826{
827 LOG_INIT (" ");

829 struct io_uring_cqe *cqe = NULL;
830 int ret = io_uring_wait_cqe (&ring, &cqe);
831 if (ret < 0)
832 {

834 LOG_END (" ");
835 return CRITICAL_CODE;
836 }
837 struct request *req = (struct request *)cqe->user_data;
838 if (!req)
839 {
840 io_uring_cqe_seen (&s->rings[0], cqe);
841 LOG_END (" ");
842 return SUCCESS_CODE;
843 }
844 if (cqe->res < 0)
845 {
846 list_delete_node (&s->list, req);
847 LOG_END (" ");
848 return CRITICAL_CODE;
849 }
850 if (req->client_socket == s->efds[0])
851 {
852 io_uring_cqe_seen (&s->rings[0], cqe);
853 list_delete_node (&s->list, req);
854 LOG_END (" ");
855 return ERROR_CODE;
856 }
857 io_uring_cqe_seen (&s->rings[0], cqe);
858 switch (req->event_type)
859 {
860 case EV_ACC:
861 handle_accept (s, cqe->res);
862 add_accept (s, client_addr, client_addr_len);
863 add_read (s, cqe->res);
864 list_delete_node (&s->list, req);
865 break;
866 case EV_REA:
867 if (cqe->res < 0)
868 {
869 handle_error (s, req);
870 }
871 if (cqe->res < 1)
872 {
873 handle_close (s, req);
874 }
875 if (cqe->res > 0)
876 {
877 handle_read (s, req);
878 }
879 list_delete_node (&s->list, req);
880 break;
881 case EV_WRI:
882 handle_write (s, req->client_socket);
883 list_delete_node (&s->list, req);
884 break;
885 }
886 LOG_END (" ");
887 return SUCCESS_CODE;
